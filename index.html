        function isIsogram(str) {
            if (str.length == 0) {
                return true
            }
            let arr = []
            let str1 = str.toLowerCase().split('').sort()
            for (let i = 0; i < str1.length; i++) {
                if (str1[i] === str1[i - 1]) {
                    arr.push(str1[i])
                }
            }
            if (arr.length == 0) {
                return true
            } else {
                return false
            }
        }
